ROOT CONFIGURATION
File: config.asm
Author: arfy slowy

OVERVIEW
Global constants and macros used throughout the kernel. This file defines fundamental sizes, shifts, masks, and ASCII constants.

PAGE CONSTANTS
  KERNEL_PAGE_mask        = 0xF000 (page mask for 32-bit addressing)
  KERNEL_PAGE_SIZE_byte  = 0x1000 (4096 bytes = 4KB)
  KERNEL_PAGE_SIZE_shift = 12 (2^12 = 4096)

SHIFT CONSTANTS (multiply by powers of 2)
  STATIC_MULTIPLE_BY_2_shift      = 1 (x2)
  STATIC_MULTIPLE_BY_8_shift      = 3 (x8)
  STATIC_MULTIPLE_BY_16_shift     = 4 (x16)
  STATIC_MULTIPLE_BY_PAGE_shift  = 12 (x4096)
  
  STATIC_DIVIDE_BY_2_shift        = 1 (/2)
  STATIC_DIVIDE_BY_4_shift        = 2 (/4)
  STATIC_DIVIDE_BY_8_shift        = 3 (/8)
  STATIC_DIVIDE_BY_PAGE_shift     = 12 (/4096)

SHIFT CONSTANTS (move bits)
  STATIC_MOVE_HIGH_TO_AL_shift    = 8
  STATIC_MOVE_HIGH_TO_AX_shift   = 16
  STATIC_MOVE_HIGH_TO_EAX_shift  = 32
  STATIC_MOVE_AX_TO_HIGH_shift   = 16
  STATIC_MOVE_EAX_TO_HIGH_shift = 32

SIZE CONSTANTS
  STATIC_BYTE_mask     = 0xFF
  STATIC_BY_SIZE_byte  = 1
  STATIC_WORD_SIZE_byte = 2
  STATIC_DWORD_SIZE_byte = 4
  STATIC_QWORD_SIZE_byte = 8
  STATIC_QWORD_SIZE_bit = 64
  STATIC_QWORD_mask = 0xFFFFFFFF00000000

SIGN BIT POSITIONS
  STATIC_BYTE_BIT_sign = 7
  STATIC_WORD_BIT_sign = 15
  STATIC_QWORD_BIT_sign = 63

ASCII CONSTANTS
  STATIC_ASCII_TERMINATOR = 0x0000
  STATIC_ASCII_BACKSPACE  = 0x0008
  STATIC_ASCII_TAB        = 0x0009
  STATIC_ASCII_NEW_LINE   = 0x000A
  STATIC_ASCII_ENTER      = 0x000D
  STATIC_ASCII_ESCAPE     = 0x001B
  STATIC_ASCII_SPACE      = 0x0020
  STATIC_ASCII_DELETE     = 0x007F

BOOLEAN CONSTANTS
  STATIC_TRUE  = 0x00
  STATIC_FALSE = 0x01
  STATIC_EMPTY = 0x00

MISC CONSTANTS
  STATIC_NOTHING         = 0x90 (NOP instruction)
  STATIC_MAX_unsigned   = -1 (0xFFFFFFFFFFFFFFFF)
  STATIC_NUMBER_SYSTEM_binary      = 2
  STATIC_NUMBER_SYSTEM_octal       = 8
  STATIC_NUMBER_SYSTEM_decimal     = 10
  STATIC_NUMBER_SYSTEM_hexadecimal = 16

STRUCTURE: STATIC_STRUCTURE_BLOCK
  Used for linked list nodes in memory allocation
  Size: 4096 bytes (one page) minus 8 bytes for link
  Fields:
    .data: 4088 bytes of usable data
    .link: 8 bytes for next node pointer
